```go
switch outcome.Type {
	case components.OutcomeTypeExit:
		// outcome.OutcomeExit is populated
	case components.OutcomeTypeTimeoutObj:
		// outcome.OutcomeTimeout is populated
	default:
		// Unknown type - use outcome.GetUnknownRaw() for raw JSON
}
```
